#### 问题描述

#### 解决方案
- 分析了现有数据流，确认 applications 表和 bookings 表之间的关联机制
- 优化了数据加载逻辑，确保两个模块都能正确显示关联数据
- 在删除应聘记录时，自动级联删除关联的预约记录

#### 修改文件
- `applications.html` - 优化数据加载和关联逻辑
- `candidate-form-complete.html` - 数据提交逻辑已优化
- `booking.html` - 预约数据管理逻辑

### 删除功能实现

```javascript
// 应聘信息删除
async function deleteApplication(appId, name) {
    if (!confirm('确定要删除应聘人 "' + name + '" 的申请记录吗？')) return;
    
    // 先删除关联的预约记录
    await db.from('bookings').delete().eq('application_id', appId);
    
    // 删除应聘记录
    await db.from('applications').delete().eq('id', appId);
    
    // 刷新列表
    loadApplications();
    loadBookings();
    loadStats();
}
```

### 排序功能实现

```javascript
// 应聘信息排序
function sortApplications(field) {
    // 切换排序方向
    if (sortField === field) {
        sortDirection = sortDirection === 'asc' ? 'desc' : 'asc';
    } else {
        sortField = field;
        sortDirection = 'asc';
    }
    
    // 排序数据
    allApplications.sort(function(a, b) {
        var valA = a[field] || '';
        var valB = b[field] || '';
        
        if (valA < valB) return sortDirection === 'asc' ? -1 : 1;
        if (valA > valB) return sortDirection === 'asc' ? 1 : -1;
        return 0;
    });
    
    // 重新渲染
    renderApplications(allApplications);
}
```

### 字段排序实现

```javascript
// 字段上移/下移
async function moveField(fieldId, direction) {
    // 找到当前字段索引
    var currentIndex = currentFieldsData.findIndex(f => f.id == fieldId);
    var newIndex = currentIndex + direction;
    
    // 交换位置
    var temp = currentFieldsData[currentIndex];
    currentFieldsData[currentIndex] = currentFieldsData[newIndex];
    currentFieldsData[newIndex] = temp;
    
    // 更新排序值
    for (var i = 0; i < currentFieldsData.length; i++) {
        currentFieldsData[i].sort_order = i + 1;
    }
    
    // 保存到数据库
    for (var i = 0; i < currentFieldsData.length; i++) {
        await db.from('field_configs').update({ 
            sort_order: currentFieldsData[i].sort_order 
        }).eq('id', currentFieldsData[i].id);
    }
    
    // 更新本地存储
    localStorage.setItem('templateFields_' + templateId, JSON.stringify(currentFieldsData));
    
    // 重新渲染
    renderFieldsList(fieldsList, currentFieldsData);
}
```

### 招聘渠道补充信息实现

```javascript
// 处理下拉框变化
function handleSelectChange(selectElement, fieldName) {
    var selectedValue = selectElement.value;
    var fieldId = selectElement.id;
    var extraDiv = document.getElementById(fieldId + '_extra');
    var extraInput = document.getElementById(fieldId + '_extra_input');
    
    if (fieldName === 'source_channel') {
        if (selectedValue === '内部介绍') {
            extraDiv.style.display = 'block';
            extraInput.placeholder = '请输入推荐人姓名';
            extraInput.required = true;
        } else if (selectedValue === '其他') {
            extraDiv.style.display = 'block';
            extraInput.placeholder = '请说明具体来源';
            extraInput.required = true;
        } else {
            extraDiv.style.display = 'none';
            extraInput.value = '';
            extraInput.required = false;
        }
    }
}
```
