        function testSDK() {
            updateStatus('正在加载 SDK...');
            
            const cdnList = [
                'https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2/dist/umd/supabase.min.js',
                'https://unpkg.com/@supabase/supabase-js@2/dist/umd/supabase.min.js'
            ];
            
            let index = 0;
            
            function tryLoad() {
                if (index >= cdnList.length) {
                    updateStatus('SDK 加载失败，请检查网络', true);
                    return;
                }
                
                const script = document.createElement('script');
                script.src = cdnList[index];
                script.onload = function() {
                    updateStatus('SDK 加载成功！');
                    testDB();
                };
                script.onerror = function() {
                    index++;
                    tryLoad();
                };
                document.head.appendChild(script);
            }
            
            tryLoad();
        }
        
        function testDB() {
            updateStatus('正在初始化数据库...');
            
            if (window.supabase && window.supabase.createClient) {
                try {
                    db = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
                    updateStatus('数据库初始化成功！');
                } catch (error) {
                    updateStatus('数据库初始化失败: ' + error.message, true);
                }
            } else {
                updateStatus('SDK 未加载', true);
            }
        }
