## 一、项目概述

### 1.1 项目背景

当前义齿工厂招聘系统已具备基础的应聘信息管理和面试预约功能，但缺乏完整的招聘流程管理模块。为实现从初试到录用的全流程数字化管理，需要开发招聘流程管理模块。

### 1.2 项目目标

构建一个完整的招聘流程管理模块，实现：
- ✅ 招聘流程全环节管理（初试、复试、录用、报到）
- ✅ 各环节数据流转与状态追踪
- ✅ 数据看板可视化呈现
- ✅ 数据筛选、导出及打印功能

### 1.3 实施范围

| 模块 | 功能 | 优先级 |
|------|------|--------|
| 数据库设计 | 招聘流程表结构设计 | 高 |
| 招聘流程管理 | 初试、复试、录用、报到全流程管理 | 高 |
| 数据看板 | 统计图表、转化率分析 | 高 |
| 数据导出 | Excel/PDF导出、打印 | 中 |
| 字段自定义 | 列表字段显示配置 | 低 |

**1. 数据加载**

```javascript
class RecruitmentDataManager {
    constructor(supabaseClient) {
        this.client = supabaseClient;
        this.allData = [];
        this.filteredData = [];
    }
    
    async loadData() {
        const { data, error } = await this.client
            .from('recruitment_process_integrated')
            .select('*')
            .order('created_at', { ascending: false });
        
        if (error) throw error;
        
        this.allData = data || [];
        this.filteredData = [...this.allData];
        return this.allData;
    }
    
    applyFilters(filters) {
        this.filteredData = this.allData.filter(item => {
            if (filters.position && item.position !== filters.position) return false;
            if (filters.job_type && item.job_type !== filters.job_type) return false;
            if (filters.stage && item.current_stage !== filters.stage) return false;
            if (filters.status && item.current_status !== filters.status) return false;
            if (filters.search) {
                const searchLower = filters.search.toLowerCase();
                if (!item.name.toLowerCase().includes(searchLower) &&
                    !item.phone.includes(searchLower)) return false;
            }
            return true;
        });
        return this.filteredData;
    }
}
```

**2. 动态表格渲染**

```javascript
class DynamicTableRenderer {
    constructor(container, fieldConfig) {
        this.container = container;
        this.fieldConfig = fieldConfig;
    }
    
    render(data, displayFields) {
        const table = document.createElement('table');
        table.className = 'recruitment-table';
        
        const thead = this.renderHeader(displayFields);
        const tbody = this.renderBody(data, displayFields);
        
        table.appendChild(thead);
        table.appendChild(tbody);
        
        this.container.innerHTML = '';
        this.container.appendChild(table);
    }
    
    renderHeader(fields) {
        const thead = document.createElement('thead');
        const tr = document.createElement('tr');
        
        tr.innerHTML = `
            <th class="checkbox-col"><input type="checkbox" id="selectAll"></th>
            ${fields.map(field => `
                <th style="width:${field.width}px" 
                    class="${field.fixed ? 'fixed-col' : ''}">
                    ${field.label}
                </th>
            `).join('')}
        `;
        
        thead.appendChild(tr);
        return thead;
    }
    
    renderBody(data, fields) {
        const tbody = document.createElement('tbody');
        
        data.forEach((row, index) => {
            const tr = document.createElement('tr');
            tr.dataset.id = row.id;
            
            tr.innerHTML = `
                <td><input type="checkbox" class="row-checkbox" value="${row.id}"></td>
                ${fields.map(field => {
                    if (field.isAction) {
                        return `<td class="action-col">${this.renderActions(row)}</td>`;
                    }
                    return `<td>${this.formatValue(row[field.field], field)}</td>`;
                }).join('')}
            `;
            
            tbody.appendChild(tr);
        });
        
        return tbody;
    }
    
    formatValue(value, field) {
        if (value === null || value === undefined) return '-';
        
        switch (field.format) {
            case 'phone':
                return this.maskPhone(value);
            case 'stage':
                return this.translateStage(value);
            case 'status':
                return this.translateStatus(value);
            case 'datetime':
                return new Date(value).toLocaleString('zh-CN');
            default:
                return value;
        }
    }
    
    maskPhone(phone) {
        return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2');
    }
    
    translateStage(stage) {
        const map = {
            'first_interview': '初试',
            'second_interview': '复试',
            'hired': '录用',
            'onboarded': '已报到'
        };
        return map[stage] || stage;
    }
    
    translateStatus(status) {
        const map = {
            'pending': '待处理',
            'pass': '通过',
            'reject': '不通过'
        };
        return map[status] || status;
    }
    
    renderActions(row) {
        const actions = {
            'first_interview': '<button class="btn-process">处理</button>',
            'second_interview': '<button class="btn-process">处理</button>',
            'hired': '<button class="btn-process">报到登记</button>'
        };
        
        return `
            <button class="btn-view" data-id="${row.id}">查看</button>
            ${actions[row.current_stage] || ''}
        `;
    }
}
```

```javascript
class DetailModal {
    constructor() {
        this.modal = document.getElementById('detailModal');
        this.content = document.getElementById('detailContent');
    }
    
    show(record) {
        const groups = RECRUITMENT_FIELD_CONFIG.detail;
        let html = '';
        
        for (const [groupName, fields] of Object.entries(groups)) {
            const groupData = fields.filter(f => record[f.field]);
            if (groupData.length === 0) continue;
            
            html += `<div class="detail-group">
                <h4>${this.translateGroupName(groupName)}</h4>
                <div class="detail-grid">`;
            
            groupData.forEach(field => {
                html += `
                    <div class="detail-item">
                        <label>${field.label}</label>
                        <span>${this.formatValue(record[field.field])}</span>
                    </div>
                `;
            });
            
            html += '</div></div>';
        }
        
        this.content.innerHTML = html;
        this.modal.style.display = 'flex';
    }
    
    hide() {
        this.modal.style.display = 'none';
    }
    
    translateGroupName(name) {
        const map = {
            'basic': '基本信息',
            'application': '应聘信息',
            'firstInterview': '初试信息',
            'secondInterview': '复试信息',
            'onboarding': '报到信息',
            'system': '系统信息'
        };
        return map[name] || name;
    }
}
```

```javascript
                data.current_stage = 'second_interview';
                data.current_status = 'pending';
            } else {
                data.current_status = 'reject';
            }
            
            await this.submit(record.id, data);
        });
        
        modal.querySelector('.btn-cancel').addEventListener('click', () => {
            modal.style.display = 'none';
        });
    }
    
    async submit(id, data) {
        const { error } = await supabase
            .from('recruitment_process_integrated')
            .update(data)
            .eq('id', id);
        
        if (error) {
            alert('提交失败：' + error.message);
            return;
        }
        
        alert('提交成功');
        location.reload();
    }
}
```

```javascript
    bindEvents(modal, data) {
        modal.querySelector('.btn-export').addEventListener('click', () => {
            const scope = modal.querySelector('input[name="exportScope"]:checked').value;
            const fields = modal.querySelector('input[name="exportFields"]:checked').value;
            const format = modal.querySelector('input[name="exportFormat"]:checked').value;
            const fileName = modal.querySelector('input[name="fileName"]').value;
            
            let exportData = data;
            
            if (scope === 'filtered') {
                exportData = currentFilteredData;
            } else if (scope === 'selected') {
                exportData = selectedRecords;
            }
            
            if (format === 'xlsx') {
                this.exportToExcel(exportData, fields, fileName);
            } else {
                this.exportToCSV(exportData, fields, fileName);
            }
            
            modal.style.display = 'none';
        });
        
        modal.querySelector('.btn-cancel').addEventListener('click', () => {
            modal.style.display = 'none';
        });
    }
    
    exportToExcel(data, fieldType, fileName) {
        const fields = fieldType === 'all' 
            ? this.fieldConfig.export 
            : this.fieldConfig.core;
        
        const wsData = [fields.map(f => f.label)];
        
        data.forEach(row => {
            wsData.push(fields.map(f => this.formatValue(row[f.field])));
        });
        
        const ws = XLSX.utils.aoa_to_sheet(wsData);
        ws['!cols'] = fields.map(() => ({ wch: 15 }));
        
        const wb = XLSX.utils.book_new();
        XLSX.utils.book_append_sheet(wb, ws, '招聘流程数据');
        
        XLSX.writeFile(wb, `${fileName}.xlsx`);
    }
    
    formatValue(value) {
        if (value === null || value === undefined) return '';
        
        if (typeof value === 'boolean') {
            return value ? '是' : '否';
        }
        
        if (value instanceof Date || /\d{4}-\d{2}-\d{2}T/.test(value)) {
            return new Date(value).toLocaleString('zh-CN');
        }
        
        return String(value);
    }
}
```

```javascript
class DashboardStats {
    calculate(data) {

            total: data.length,

        ).length;

        const secondTotal = data.filter(d =>
```
